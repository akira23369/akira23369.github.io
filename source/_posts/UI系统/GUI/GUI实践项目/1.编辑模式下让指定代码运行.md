# 在类名上添加 `[ExecuteAlways]` 特性可以让指定代码在编辑模式下运行
编辑模式下，缩放Game窗口大小每一帧相关也会执行

```cs
//在类名添加ExecuteAlways特性 可以让编辑模式下让指定代码运行
[ExecuteAlways]
public class Lesson01_必备知识点_编辑模式下让指定代码运行 : MonoBehaviour
{
    private void Awake()
    {
        Debug.Log("对象初始化时调用（即使脚本未启用）。");
    }

    private void OnEnable()
    {
        Debug.Log("脚本启用或对象激活时调用（在Awake之后，若对象初始启用）。");
    }

    void Start()
    {
        Debug.Log("在首次Update前调用（仅一次，需脚本启用）。");
    }

    private void FixedUpdate()
    {
        Debug.Log("按固定时间间隔调用（物理计算阶段，可能一帧多次）。");
    }

    private void OnTriggerEnter(Collider other)
    {
        Debug.Log("物理触发事件，在FixedUpdate之后、Update之前处理。");
    }

    void Update()
    {
        Debug.Log("每帧调用一次。");
    }

    private void LateUpdate()
    {
        Debug.Log("所有Update完成后调用。");
    }

    private void OnGUI()
    {
        Debug.Log("渲染GUI时调用，每帧可能多次（在LateUpdate之后）。");
    }

    private void OnDisable()
    {
        Debug.Log("脚本禁用或对象失活时调用。");
    }

    private void OnDestroy()
    {
        Debug.Log("对象销毁前调用。");
    }
}
```
